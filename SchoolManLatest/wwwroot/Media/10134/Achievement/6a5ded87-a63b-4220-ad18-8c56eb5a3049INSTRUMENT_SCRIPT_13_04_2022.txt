CREATE TRIGGER [dbo].[TriggerAfterODPPMaterialsInsert]
   ON  [dbo].[TempODPMaterials]
   AFTER INSERT
AS 
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;
	declare @ODP nvarchar(12);  
	declare @Phase nvarchar(4);
	declare @MaterialCode nvarchar(18);  
	declare @Number int;
	declare @Position int;
	declare @Quantity float;  
	declare @Availability bit;
	declare @Description  nvarchar(40); 
        declare @MeasureUnit  nvarchar(3); 
        declare @ClosingPhase bit;
	declare @DATA  nvarchar(max); 
	declare @REASON  nvarchar(max); 


        select @ODP=i.ODP from inserted i; 
	select @Phase=i.Phase from inserted i; 
	select @MaterialCode=i.MaterialCode from inserted i; 
	select @Number =i.Number from inserted i; 
	select @Position =i.Position from inserted i; 
	select @Quantity =i.Quantity from inserted i; 
	select @Availability =i.Availability from inserted i; 
	select @Description =i.Description from inserted i; 
    select @MeasureUnit =i.MeasureUnit from inserted i; 
    select @ClosingPhase =i.ClosingPhase from inserted i; 
 
 IF  EXISTS (SELECT * FROM [dbo].[ODP] WHERE ODP = @ODP)---ODP FOREIGN KEY VIOLATION CHECKING
 BEGIN
	  IF NOT EXISTS (SELECT * FROM [dbo].[ODPMaterials] WHERE ODP = @ODP and Phase = @Phase and MaterialCode=@MaterialCode and Number=@Number and Position=@Position)
	  BEGIN
		insert into [dbo].[ODPMaterials]  
		(ODP,Phase,MaterialCode,Number,Position,Quantity,Availability,Description,MeasureUnit,ClosingPhase)
		values(@ODP,@Phase,@MaterialCode,@Number,@Position,@Quantity,@Availability,@Description,@MeasureUnit,@ClosingPhase); 

		delete from [TEMPODPMaterials]  WHERE ODP = @ODP and Phase = @Phase and MaterialCode=@MaterialCode and Number=@Number and Position=@Position


		Declare @OdpCount int  = (Select count(ODP) FROM [dbo].[ODPMaterials]  where ODP=@ODP)
		IF(@OdpCount >= 5) 
		BEGIN
		SET @REASON = CAST(@OdpCount AS VARCHAR(MAX)) + ' records with odp number ' + CAST(@ODP AS VARCHAR(MAX)) 
	 
	   INSERT INTO [dbo].[LogErrorData](TableName,Data,Reason,TimeStamp)
	    VALUES('[dbo].[ODPMaterials]',@DATA,@REASON,getdate())
		END

	   END
	   ELSE
	   BEGIN
		 update [dbo].[ODPMaterials]  set ODP=@ODP,Phase=@Phase,MaterialCode=@MaterialCode,Number=@Number,Position=@Position,Quantity=@Quantity,Availability=@Availability,Description=@Description,MeasureUnit=@MeasureUnit,ClosingPhase=@ClosingPhase WHERE ODP = @ODP and Phase =@Phase and MaterialCode=@MaterialCode and Number=@Number and Position=@Position
		
		   --inserting in error table 
		    SET @DATA = @ODP + ' || ' + @Phase + ' || ' + @MaterialCode + ' || ' + CAST (@Number AS NVARCHAR(10))+' || '  + CAST (@Position AS NVARCHAR(10)) + ' || ' + CAST (@Quantity AS NVARCHAR(MAX))+ ' || ' +
	           CAST(@Availability AS VARCHAR(MAX)) + ' || ' +@Description
			  

	  SET @REASON = 'Violation of primary key constraint for Key ' + @ODP + ' in table [dbo].[ODPMaterials]'
	 
	   INSERT INTO [dbo].[LogErrorData](TableName,Data,Reason,TimeStamp)
	    VALUES('[dbo].[ODPMaterials]',@DATA,@REASON,getdate())
		
		delete from [TEMPODPMaterials]  WHERE ODP = @ODP and Phase = @Phase and MaterialCode=@MaterialCode and Number=@Number and Position=@Position
	   END
 END
 ELSE
 BEGIN
      --inserting in error table 
		 SET @DATA = @ODP + ' || ' + @Phase + ' || ' + @MaterialCode + ' || ' + CAST (@Number AS NVARCHAR(10))+' || '  + CAST (@Position AS NVARCHAR(10)) + ' || ' + CAST (@Quantity AS NVARCHAR(MAX))+ ' || ' +
	           CAST(@Availability AS VARCHAR(MAX)) + ' || ' +@Description 		 
	  SET @REASON = 'Violation of foreign key constraint for Key ' + @ODP + ' in table [dbo].[ODPMaterials]'
	 
	   INSERT INTO [dbo].[LogErrorData](TableName,Data,Reason,TimeStamp)
	    VALUES('[dbo].[ODPMaterials]',@DATA,@REASON,getdate())

		delete from [TEMPODPMaterials]  WHERE ODP = @ODP and Phase = @Phase and MaterialCode=@MaterialCode and Number=@Number and Position=@Position
 END
